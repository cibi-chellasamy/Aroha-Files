INPUT:

COUNTRY 	SALES
IND	   	1000
IND	   	2000
IND	   	3000
UK	   	4400
USA	   	500
IND	  	1000

OUTPUT:

COUNTRY     SALES      %SALES
IND	       7000	         58.82
USA	       500	          4.20
UK	       4400	         36.97

QUERY:

SELECT DISTINCT COUNTRY,
SUM(SALES) OVER (PARTITION BY COUNTRY) "SALES",
(SUM(SALES) OVER (PARTITION BY COUNTRY)/SUM(SALES) OVER()) *100 "%SALES"
FROM INPUT;
