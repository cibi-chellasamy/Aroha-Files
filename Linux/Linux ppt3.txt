1. grep :: to search for particular string/word in a text file
           syntax : grep string/word file_name
                    grep -i string/word file_name ( case insensitive strings/word)
                    grep -n string/word file_name ( along with line numbers )
                     
=====================================================================================================

2. | :: output the result of one command as input to another command 
        ls -l | grep word/string filename

=====================================================================================================

                         File type --- - ( normal file ), d ( directory file ), s ( socket file ), l (link file)
                         File permissions --- read,write,execute for user(root),group,others
                         Number of links 
                         Owner -- owner of the file
                         Group -- group of he file
                         Size -- size in bytes
                         Last modified date and time 
                         File name

3.Users permissions :: read (4), write (2), execute (1)
                       user (u), group (g), others (o), all (a)

                       chmod : to change access permissions of files and directories
                               eg : chmod g+wx filename
                                    chmod u=rwx,o-wx filename
                                    chmod 755 filename rwx == r 4 + w 2 + x 1 = 7 ( user)
                                                       rwx == r 4 + x 1 = 5 ( group)
                                                       rwx == r 4 + x 1 = 5 ( others )
                                                 
                       chown : to change the owner of files and directories
                               eg : chown username filename
                                    chown username:groupname filename

                       chgrp : to change the group ownership of files and directories 
			       eg : chgrp groupname filename      

=========================================================================================================

4. less :: to read the starting content from a very big data
           Syntax : less file_name
                    and press down arrow key ( to view the content downwards )

=========================================================================================================

5.top :: view of the running system 
         Syntax : top ( it gets refreshed after ever 3s )

========================================================================================================

6.head :: it outputs the first part of the file (usually 10lines )
          Syntax : head file_name
                   head -nhow_many_lines filename

=======================================================================================================    

7.tail :: it outputs the last part of the file
          Syntax : tail -nhow_many_lines file_name

=========================================================================================================

8.wc :: word count ( prints the count of each word, lines etc )
        wc filename
             number_of_lines number_of_words number_of_characters filename
              (wc -l fname )  (wc -w fname)	(wc -c fname)	

=========================================================================================================
9.mv :: 1) Renaming a file or directory.
        2) Moving a file or directory to another location.
        mv source_file_name destination_file_name : to rename a file
        mv source_file_name destination_path : to move a file

==========================================================================================================
10.cp :: copy files.
        syntax : cp source_file destination_file
                 cp -i source_file destination_file ( confirmation )
        copy multiple files to directory : cp src_file1 src_file2 src_file3 dest_directory_name/   
        copy files between two directories :  cp -R source_directory dest_directory

==========================================================================================================
11.vi :: vi file_name 
         enter insert_mode (press i )
         type the content
         press esc 
         type :wq and press enter (w for save and q for quit )
